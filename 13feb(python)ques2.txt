lass Employee:
    def __init__(self,id,name,leaves):
        self.id=id
        self.name=name
        self.leaves=leaves
 
class Company:
    def __init__(self,cname,emps):
        self.cname=cname
        self.emps=emps
 
    def display_leaves(self,empid,Itype):
        for i in self.emps:
            if i.id==empid:
                for a,b in i.leaves.items():
                    if Itype==a:
                        return b
         
    def leave_application(self,empid,Itype,nol):
        for i in self.emps:
            if i.id==empid:
                for a,b in i.leaves.items():
                    if Itype==a:
                        if b>=nol:
                            return "Granted"
                        else:
                            return "Rejected"
